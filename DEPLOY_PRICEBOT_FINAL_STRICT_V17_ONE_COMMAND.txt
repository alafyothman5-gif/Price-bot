sudo bash -lc '
set -euo pipefail
APP=/opt/pricebot
cd "$APP"
TS=$(date +%Y%m%d_%H%M%S)
mkdir -p /root/pricebot_backups
if [ -d "$APP" ]; then tar -czf "/root/pricebot_backups/pricebot_before_v17_$TS.tar.gz" -C /opt pricebot --exclude="pricebot/venv" --exclude="pricebot/__pycache__" --exclude="pricebot/backups" || true; fi
python3 -m venv venv
./venv/bin/pip install --upgrade pip
./venv/bin/pip install -r requirements.txt
./venv/bin/python -m py_compile app.py admin.py database.py matcher.py matcher_v2.py matcher_v3.py matcher_v4.py product_intelligence.py
./venv/bin/python acceptance_tests_final_v17.py
cat > /etc/systemd/system/pricebot.service <<EOF
[Unit]
Description=PriceBot WhatsApp Webhook
After=network.target

[Service]
Type=simple
User=root
WorkingDirectory=/opt/pricebot
EnvironmentFile=-/opt/pricebot/.env
ExecStart=/opt/pricebot/venv/bin/python -m uvicorn app:app --host 127.0.0.1 --port 8000
Restart=always
RestartSec=3

[Install]
WantedBy=multi-user.target
EOF
systemctl daemon-reload
systemctl restart pricebot
sleep 5
systemctl status pricebot --no-pager -l | sed -n "1,35p"
curl -kfsS https://46.101.148.246.sslip.io/health || true
journalctl -u pricebot -n 80 --no-pager
'
