# بعد رفع ZIP إلى السيرفر داخل /root مثلاً، شغّل هذا الأمر من التيرمنل:
cd /opt/pricebot && sudo cp -a /opt/pricebot "/root/pricebot_BACKUP_BEFORE_CATALOG_ENRICHMENT_$(date +%Y%m%d_%H%M%S)" && sudo unzip -o /root/PriceBot_Catalog_Enrichment_READY.zip -d /opt/pricebot && sudo chown -R root:root /opt/pricebot && python3 -m pip install -r requirements.txt && python3 -m py_compile app.py database.py matcher.py matcher_v2.py matcher_v3.py matcher_v4.py admin.py tools/enrich_products_catalog.py && python3 -m unittest discover -s tests -v && sudo systemctl restart pricebot && sudo systemctl status pricebot --no-pager -l
