sudo bash -lc '
set -e
APP=/opt/pricebot
REPO="https://github.com/alafyothman5-gif/Price-bot.git"
BRANCH="${BRANCH:-main}"
STAMP=$(date +%Y%m%d_%H%M%S)
TMP="/tmp/pricebot_deploy_$STAMP"
BACKUP="/root/pricebot_backup_before_ready_$STAMP.tar.gz"

echo "1) Backup current project..."
tar -czf "$BACKUP" -C /opt pricebot --exclude="pricebot/venv" --exclude="pricebot/__pycache__" || true

echo "2) Clone latest GitHub code..."
rm -rf "$TMP"
git clone --depth 1 --branch "$BRANCH" "$REPO" "$TMP"

echo "3) Preserve live secrets/database..."
mkdir -p /root/pricebot_preserve_$STAMP
[ -f "$APP/.env" ] && cp "$APP/.env" /root/pricebot_preserve_$STAMP/.env
[ -f "$APP/pricebot.db" ] && cp "$APP/pricebot.db" /root/pricebot_preserve_$STAMP/pricebot.db

echo "4) Install files..."
rsync -a --delete --exclude ".git" --exclude ".env" --exclude "pricebot.db" --exclude "*.db" --exclude "venv" --exclude "__pycache__" --exclude "backups" "$TMP"/ "$APP"/
[ -f /root/pricebot_preserve_$STAMP/.env ] && cp /root/pricebot_preserve_$STAMP/.env "$APP/.env"
[ -f /root/pricebot_preserve_$STAMP/pricebot.db ] && cp /root/pricebot_preserve_$STAMP/pricebot.db "$APP/pricebot.db"

cd "$APP"
python3 -m venv venv
./venv/bin/python -m pip install --upgrade pip
./venv/bin/pip install -r requirements.txt
find . -path "./venv" -prune -o -name "*.py" -print0 | xargs -0 ./venv/bin/python -m py_compile
./venv/bin/python acceptance_tests_v4.py || true

PORT=$(grep -RhoE "127.0.0.1:[0-9]+|localhost:[0-9]+" /etc/caddy 2>/dev/null | grep -oE "[0-9]+" | head -n1 || true)
PORT=${PORT:-8000}
cat > /etc/systemd/system/pricebot.service <<EOF
[Unit]
Description=PriceBot WhatsApp Webhook
After=network.target

[Service]
Type=simple
User=root
WorkingDirectory=/opt/pricebot
EnvironmentFile=-/opt/pricebot/.env
Environment=PRICEBOT_BACKGROUND_WARMUP=1
ExecStart=/opt/pricebot/venv/bin/python -m uvicorn app:app --host 127.0.0.1 --port $PORT
Restart=always
RestartSec=3

[Install]
WantedBy=multi-user.target
EOF

systemctl daemon-reload
systemctl reset-failed pricebot
systemctl enable pricebot >/dev/null
systemctl restart pricebot
sleep 5
systemctl status pricebot --no-pager -l | sed -n "1,35p"
echo "LOCAL HEALTH:"
curl -fsS http://127.0.0.1:$PORT/health || true
echo "PUBLIC HEALTH:"
curl -kfsS https://46.101.148.246.sslip.io/health || true
echo "LOGS:"
journalctl -u pricebot -n 80 --no-pager
'
